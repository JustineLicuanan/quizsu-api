Quizsu API Endpoints

  • # Create quiz endpoint

    POST /quiz/create

    Content-Type: application/json

    {
  "title": "True or False",
  "password": "ultimateQuizPass",
  "questions": [
      {
        "text":
          "Are you a developer?",
        "answers": ["true", "t"]
      },
      {
        "text":
          "Do you have a lovelife?",
        "answers": ["false", "f"]
      }
  ]
}

  • # Get quiz results endpoint

    POST /quiz/results/${id}

    Content-Type: application/json

    {
  "password": "ultimateQuizPass"
}

  • # Take quiz endpoint

    GET /quiz/take/${id}

  • # Pass quiz endpoint

    POST /quiz/pass/${id}

    Content-Type: application/json

    {
  "name": "John Doe",
  "answers": [
      {
        "questionId":
          "fromTakeQuizEndpoint",
        "answer": "t"
      },
      {
        "questionId":
          "fromTakeQuizEndpoint",
        "answer": "false"
      }
  ]
}

  • # Disable quiz endpoint

    PATCH /quiz/disable/${id}

    Content-Type: application/json

    {
  "password": "ultimateQuizPass"
}
